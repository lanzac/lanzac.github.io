\newcommand{\resumeHeading}[4]{
    \begin{tabular*}{0.99\textwidth}[t]{l@{\extracolsep{\fill}}r}

    \end{tabular*}\vspace{-5pt}
}

\newcommand{\resumeSubheadingBis}[4]{

    \end{tabular*}

    \vspace{2pt}
}

      { {{- entry.institution -}} }{ {{- entry.studyType }} in {{ entry.area -}} }{ {{ entry.endDate -}} }
